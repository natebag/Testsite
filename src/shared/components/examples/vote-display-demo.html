6. Integration Examples

Code examples showing how to integrate the vote display system with existing MLG.clan components.

JavaScript Integration


// Initialize the vote display system
import { VoteDisplaySystem, VoteDisplayUtils } from './vote-display-ui.js';
import { SolanaVotingSystem } from '../voting/solana-voting-system.js';

// Create voting system instance
const votingSystem = new SolanaVotingSystem();
await votingSystem.initialize(wallet);

// Create vote display system
const voteDisplay = await VoteDisplayUtils.createVoteDisplaySystem(
  votingSystem, 
  wallet,
  {
    onVoteUpdated: (updates) => console.log('Votes updated:', updates),
    onNetworkStatusChange: (status) => console.log('Network:', status)
  }
);

// Add compact vote display to content tile
const contentTile = document.querySelector('.content-tile');
const voteDisplay = voteDisplay.createCompactVoteDisplay('clip-123', {
  showFreeVotes: true,
  showMLGVotes: true,
  enableVoting: true
});
contentTile.appendChild(voteDisplay);
